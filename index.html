Loop Control & Array Operations

Approach: Loop from 2 up to num / 2. If a factor is found, flag the number as composite and use the break statement to immediately terminate the loop early, preventing unnecessary calculations.

#include <stdio.h>
#include <conio.h>

void main() {
    int num, i, isPrime = 1;
    clrscr();

    printf("Enter a positive integer: ");
    scanf("%d", &num);

    if (num <= 1) {
        isPrime = 0;
    } else {
        for (i = 2; i <= num / 2; i++) {
            if (num % i == 0) {
                isPrime = 0; /* Number is divisible, not prime */
                break;       /* Exit loop immediately */
            }
        }
    }

    if (isPrime == 1) {
        printf("%d is a Prime Number.\n", num);
    } else {
        printf("%d is not a Prime Number.\n", num);
    }

    getch();
}

Approach: The continue statement skips the remaining code inside the loop for the current iteration and jumps directly to the next iteration. This program uses it to print numbers from 1 to 10 except for 5.

#include <stdio.h>
#include <conio.h>

void main() {
    int i;
    clrscr();

    printf("Printing numbers from 1 to 10 (skipping 5):\n");
    for (i = 1; i <= 10; i++) {
        if (i == 5) {
            continue; /* Skip the rest of the loop block for i = 5 */
        }
        printf("%d ", i);
    }
    printf("\n");

    getch();
}

Approach: Create a fixed-size array, take input values from the user using a loop, and add each element one by one to a running cumulative sum variable tracker.

#include <stdio.h>
#include <conio.h>

void main() {
    int arr[50], size, i, sum = 0;
    clrscr();

    printf("Enter the size of the array: ");
    scanf("%d", &size);

    printf("Enter %d elements:\n", size);
    for (i = 0; i < size; i++) {
        printf("Element %d: ", i + 1);
        scanf("%d", &arr[i]);
    }

    /* Addition logic */
    for (i = 0; i < size; i++) {
        sum = sum + arr[i];
    }

    printf("\nSum of all elements in the array = %d\n", sum);

    getch();
}

Approach: Assume the first element of the array is the smallest. Loop through the remaining elements and update your small tracker variable whenever a lesser value is encountered.

#include <stdio.h>
#include <conio.h>

void main() {
    int arr[50], size, i, small;
    clrscr();

    printf("Enter array size: ");
    scanf("%d", &size);

    printf("Enter %d elements:\n", size);
    for (i = 0; i < size; i++) {
        scanf("%d", &arr[i]);
    }

    /* Start comparison assuming index 0 is smallest */
    small = arr[0];

    for (i = 1; i < size; i++) {
        if (arr[i] < small) {
            small = arr[i];
        }
    }

    printf("\nThe smallest element in the array is: %d\n", small);

    getch();
}

Approach: Accept three distinct numeric entries from the user and evaluate their boundaries using nested or logical AND condition checks.

#include <stdio.h>
#include <conio.h>

void main() {
    int a, b, c;
    clrscr();

    printf("Enter three distinct numbers: ");
    scanf("%d %d %d", &a, &b, &c);

    if (a >= b && a >= c) {
        printf("%d is the largest number.\n", a);
    } else if (b >= a && b >= c) {
        printf("%d is the largest number.\n", b);
    } else {
        printf("%d is the largest number.\n", c);
    }

    getch();
}
