Chapter 1: Basic Mathematical & Logic Programs

Approach: Declare integer variables, accept user input via scanf(), process their arithmetic sum, and output the computed total.

#include <stdio.h>
#include <conio.h>

void main() {
    int num1, num2, sum;
    clrscr();

    printf("Enter two integers: ");
    scanf("%d %d", &num1, &num2);

    sum = num1 + num2;

    printf("Sum = %d\n", sum);

    getch();
}

Approach: Track sequential additions by summing the two preceding terms and shifting variable values on each loop iteration.

#include <stdio.h>
#include <conio.h>

void main() {
    int terms, i, t1 = 0, t2 = 1, nextTerm;
    clrscr();

    printf("Enter the number of terms: ");
    scanf("%d", &terms);

    printf("Fibonacci Series: ");

    for (i = 1; i <= terms; i++) {
        printf("%d ", t1);
        nextTerm = t1 + t2;
        t1 = t2;
        t2 = nextTerm;
    }
    printf("\n");

    getch();
}

Chapter 4: Advanced Loop Control & Array Operations

Approach: Test for factors from 2 up to num / 2. If a factor is found, change the flag and use a break statement to terminate the loop early.

#include <stdio.h>
#include <conio.h>

void main() {
    int num, i, isPrime = 1;
    clrscr();

    printf("Enter a positive integer: ");
    scanf("%d", &num);

    if (num <= 1) {
        isPrime = 0;
    } else {
        for (i = 2; i <= num / 2; i++) {
            if (num % i == 0) {
                isPrime = 0;
                break;       /* Exit loop immediately */
            }
        }
    }

    if (isPrime == 1) {
        printf("%d is a Prime Number.\n", num);
    } else {
        printf("%d is not a Prime Number.\n", num);
    }

    getch();
}

Approach: Use the continue statement to skip the remaining code within the current loop iteration. This example skips printing the number 5.

#include <stdio.h>
#include <conio.h>

void main() {
    int i;
    clrscr();

    printf("Printing numbers from 1 to 10 (skipping 5):\n");
    for (i = 1; i <= 10; i++) {
        if (i == 5) {
            continue; /* Jumps directly to the next loop iteration */
        }
        printf("%d ", i);
    }
    printf("\n");

    getch();
}

Approach: Store elements in an array, read inputs from the user, and use a loop to accumulate the values into a sum variable tracker.

Approach: Initialize a tracking variable with the first element of the array. Iterate through the array and update the minimum value whenever a smaller element is found.

#include <stdio.h>
#include <conio.h>

void main() {
    int arr[50], size, i, small;
    clrscr();

    printf("Enter array size: ");
    scanf("%d", &size);

    printf("Enter %d elements:\n", size);
    for (i = 0; i < size; i++) {
        scanf("%d", &arr[i]);
    }

    small = arr[0];

    for (i = 1; i < size; i++) {
        if (arr[i] < small) {
            small = arr[i];
        }
    }

    printf("\nThe smallest element in the array is: %d\n", small);

    getch();
}
