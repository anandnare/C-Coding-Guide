#include <conio.h>

Approach: Evaluate a score value across multi-level logic branches to categorize custom grade assignments cleanly.

#include <stdio.h>
#include <conio.h>

void main() {
    int marks;
    clrscr();

    printf("Enter student marks (0-100): ");
    scanf("%d", &marks);

    if (marks >= 80 && marks <= 100) {
        printf("Grade: A\n");
    } else if (marks >= 60 && marks < 80) {
        printf("Grade: B\n");
    } else if (marks >= 40 && marks < 60) {
        printf("Grade: C\n");
    } else if (marks >= 0 && marks < 40) {
        printf("Grade: Fail\n");
    } else {
        printf("Invalid Marks Entered!\n");
    }

    getch();
}

Approach: Accumulate the product total from 1 to N, storing values inside a long int variable to prevent data overflow limits.

#include <stdio.h>
#include <conio.h>

void main() {
    int num, i;
    long int fact = 1;
    clrscr();

    printf("Enter a number: ");
    scanf("%d", &num);

    if (num < 0) {
        printf("Factorial of negative numbers does not exist.\n");
    } else {
        for(i = 1; i <= num; i++) {
            fact = fact * i;
        }
        printf("Factorial of %d = %ld\n", num, fact);
    }

    getch();
}

Approach: Track sequential additions by swapping indices across historical iteration pairs inside tracking registers.

#include <stdio.h>
#include <conio.h>

void main() {
    int terms, i, t1 = 0, t2 = 1, nextTerm;
    clrscr();

    printf("Enter the number of terms: ");
    scanf("%d", &terms);

    printf("Fibonacci Series: ");

    for (i = 1; i <= terms; i++) {
        printf("%d ", t1);
        nextTerm = t1 + t2;
        t1 = t2;
        t2 = nextTerm;
    }
    printf("\n");

    getch();
}
